from scipy.optimize import linprog

def solve_lineal(c, A, b):
    result = linprog(c, A_ub=A, b_ub=b, bounds=(0, None))
    print(f"Valor Ã³ptimo: {result.fun}")
    print(f"Valores de las variables: {result.x}\n")

def ejercicio_0():
    print("Ejercicio 0")
    c = [-3, -2]
    A = [[1, 1], 
         [2, 1]]
    b = [4, 5]
    solve_lineal(c, A, b)
